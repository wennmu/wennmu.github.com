    # 修改服务器时间

    #查看服务器日期时间

date --set 08:02:00

    [root@admin ~]# date
Tue Jun  8 10:17:30 CST 2021
[root@admin ~]# date --set 08:02:00
Tue Jun  8 08:02:00 CST 2021
[root@admin ~]# date
Tue Jun  8 08:02:01 CST 2021
